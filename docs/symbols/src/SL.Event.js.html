303         triggerEvent: function (event, data, elem, onlyHandlers) {
304             /// <summary>
305             /// 
306             /// </summary>
307             /// <param name="event"></param>
308             /// <param name="data"></param>
309             /// <param name="elem"></param>
310             /// <param name="onlyHandlers">只在 .triggerHandler用到了，即不触发元素的默认行为，且停止冒泡。</param>
311             /// <returns type=""></returns>
312             if (elem && (elem.nodeType === 3 || elem.nodeType === 8)) {
313                 return;
314             }
315             var type = event.type || event,
316 			namespaces = [],
317 			cache, exclusive, i, cur, old, ontype, special, handle, eventPath, bubbleType;
318 
319             // 仅对focus/blur事件变种成focusin/out进行处理
320             // 如果浏览器原生支持focusin/out，则确保当前不触发他们
321             //预留判断 目前特殊事件没实现
322             if (rfocusMorph.test(type + EventOperator.triggered)) {
323                 return;
324             }
325 
326             if (type.indexOf("!") >= 0) {
327                 //如果类型中包含！表示触发没有包含命名空间的事件
328                 type = type.slice(0, -1);
329                 exclusive = true;
330             }
331             //包含命名空间
332             if (type.indexOf(".") >= 0) {
333                 namespaces = type.split(".");
334                 type = namespaces.shift();
335                 namespaces.sort();
336             }
337             //如果从来没有绑定过此种事件，也不用继续执行了
338             if (!elem && !EventOperator.global[type]) {
339                 return;
340             }
341 
342             // Caller can pass in an Event, Object, or just an event type string
343             event = typeof event === "object" ?
344 			event[sl.expando] ? event :
345 			new SL.Event(type, event) :
346 			new SL.Event(type);
347 
348             //判断命名空间
349             event.type = type;
350             event.isTrigger = true;
351             event.exclusive = exclusive;
352             event.namespace = namespaces.join(".");
353             event.namespace_re = event.namespace ? new RegExp("(^|\\.)" + namespaces.join("\\.(?:.*\\.)?") + "(\\.|$)") : null;
354             ontype = type.indexOf(":") < 0 ? "on" + type : "";
355 
356             event.result = undefined;
357             if (!event.target) {
358                 event.target = elem;
359             }
360             data = data != null ? sl.Convert.convertToArray(data) : [];
361             data.unshift(event);
362 
363 
364 
365             //铺设往上冒泡的路径，每小段都包括处理对象与事件类型
366             eventPath = [[elem, type]];
367             if (!onlyHandlers && !sl.InstanceOf.Window(elem)) {
368                 // 冒泡时是否需要转成别的事件(用于事件模拟)
369                 // 如果不是变形来的foucusin/out事件
370                 bubbleType = type; //预留接口
371                 cur = rfocusMorph.test(bubbleType + type) ? elem : elem.parentNode;
372                 for (old = elem; cur; cur = cur.parentNode) {
373                     eventPath.push([cur, bubbleType]);
374                     old = cur;
375                 }
376 
377                 //一直冒泡到window
378                 if (old === (elem.ownerDocument || document)) {
379                     eventPath.push([old.defaultView || old.parentWindow || window, bubbleType]);
380                 }
381             }
382 
383             //沿着之前铺好的路触发事件
384             for (i = 0; i < eventPath.length && !event.isPropagationStopped; i++) {
385 
386                 cur = eventPath[i][0];
387                 event.type = eventPath[i][1];
388 
389                 handle = (sl.data(cur, "events") || {})[event.type] && sl.data(cur, "handle");
390                 if (handle) {
391                     handle.apply(cur, data);
392                 }
393                 //一直冒泡到window
394                 handle = ontype && cur[ontype];
395                 if (handle && handle.apply(cur, data) === false) {
396                     event.preventDefault();
397                 }
398             }
399             event.type = type;
400 
401             //不触发元素默认行为
402             if (!onlyHandlers && !event.isDefaultPrevented) {
403 
404                 var isClick = elem.nodeName == "A" && type === "click";
405                 if (!isClick) {
406                     // window不触发默认工作
407                     //<ie9 focus blur对隐藏元素不触发默认动作
408                     if (ontype && elem[type] && ((type !== "focus" && type !== "blur") || event.target.offsetWidth !== 0) && !sl.InstanceOf.Window(elem)) {
409                         /* 假设type为click
410                         因为下面想通过click()来触发默认操作，
411                         但是又不想执行对应的事件处理器（re-trigger），
412                         所以需要做两方面工作：
413                         首先将elem.onclick = null；
414                         然后将EventOperator.triggered = 'click'; 
415                         将在入口handle（第62行）不再触发了
416                         之后再将它们还原*/
417                         old = elem[ontype];
418 
419                         if (old) {
420                             elem[ontype] = null;
421                         }
422 
423 
424                         EventOperator.triggered = type;
425                         elem[type]();
426                         EventOperator.triggered = undefined;
427 
428                         if (old) {
429                             elem[ontype] = old;
430                         }
431                     }
432                 }
433 
434             }
435             return event.result;

541

551                         }
552                     }
553                 }
554             }
