 10     function offset() {
 11         this.init = function () {
 12             if (this.Initialed) return;
 13             var body = document.body, container = document.createElement("div"), innerDiv, checkDiv, table, td, bodyMarginTop = parseFloat(sl.css(body, "marginTop")) || 0,
 14 			html = "<div style='position:absolute;top:0;left:0;margin:0;border:5px solid #000;padding:0;width:1px;height:1px;'><div></div></div><table style='position:absolute;top:0;left:0;margin:0;border:5px solid #000;padding:0;width:1px;height:1px;' cellpadding='0' cellspacing='0'><tr><td></td></tr></table>";
 15 
 16             sl.extend(container.style, { position: "absolute", top: 0, left: 0, margin: 0, border: 0, width: "1px", height: "1px", visibility: "hidden" });
 17 
 18             container.innerHTML = html;
 19             body.insertBefore(container, body.firstChild);
 20             innerDiv = container.firstChild;
 21             checkDiv = innerDiv.firstChild;
 22             td = innerDiv.nextSibling.firstChild.firstChild;
 23 
 24             //定位的时候 是否加上offsetParent的边框width
 25             this.doesNotAddBorder = (checkDiv.offsetTop !== 5);
 26             //定位的时候  rd是否已经加上边框
 27             this.doesAddBorderForTableAndCells = (td.offsetTop === 5);
 28 
 29             checkDiv.style.position = "fixed", checkDiv.style.top = "20px";
 30             // safari subtracts parent border width here which is 5px
 31             //是否支持fixed
 32             this.supportsFixedPosition = (checkDiv.offsetTop === 20 || checkDiv.offsetTop === 15);
 33             checkDiv.style.position = checkDiv.style.top = "";
 34 
 35             innerDiv.style.overflow = "hidden", innerDiv.style.position = "relative";
 36             this.subtractsBorderForOverflowNotVisible = (checkDiv.offsetTop === -5);
 37             //body的offset是否包含了margin
 38             this.doesNotIncludeMarginInBodyOffset = (body.offsetTop !== bodyMarginTop);
 39 
 40             body.removeChild(container);
 41             body = container = innerDiv = checkDiv = table = td = null;
 42         };
 43         this.Initialed = false;
 44     }
 45     offset.prototype = {
 46         /**
 47         *@ignore
 48         */
 49         bodyOffset: function (body) {
 50             var top = body.offsetTop, left = body.offsetLeft;
 51 
 52             this.init();
 53 
 54             if (this.doesNotIncludeMarginInBodyOffset) {
 55                 top += parseFloat(sl.css(body, "marginTop")) || 0;
 56                 left += parseFloat(sl.css(body, "marginLeft")) || 0;
 57             }
 58 
 59             return { top: top, left: left };
 60         },
 61         /**
 62         *@ignore
 63         */
 64         getOffset: function (node) {
 65             if ("getBoundingClientRect" in document.documentElement) {
 66 
 67                 if (!node || !node.ownerDocument) {
 68                     return null;
 69                 }
 70 
 71                 if (node === node.ownerDocument.body) {
 72                     return this.bodyOffset(node);
 73                 }
 74                 var box = node.getBoundingClientRect();
 75                 if (!box || node.ownerDocument.documentElement == node) {
 76                     return box ? { top: box.top, left: box.left} : { top: 0, left: 0 };
 77                 }
 78 
 79 
 80 
 81                 var doc = node.ownerDocument, body = doc.body, docElem = doc.documentElement,
 82 			clientTop = docElem.clientTop || body.clientTop || 0, clientLeft = docElem.clientLeft || body.clientLeft || 0,
 83 			top = box.top + (self.pageYOffset || sl.Support.boxModel && docElem.scrollTop || body.scrollTop) - clientTop,
 84 			left = box.left + (self.pageXOffset || sl.Support.boxModel && docElem.scrollLeft || body.scrollLeft) - clientLeft;
 85 
 86                 return { top: top, left: left };
 87 
 88             }
 89         },
 90         /**
 91         *@ignore
 92         */
 93         setOffset: function (elem, options) {
 94             var position = sl.css(elem, "position");
 95             if (position === "static") {
 96                 elem.style.position = "relative";
 97             }
 98 
 99             var curOffset = this.getOffset(elem),
100 			curCSSTop = sl.css(elem, "top"),
101 			curCSSLeft = sl.css(elem, "left"),
102 			calculatePosition = (position === "absolute" || position === "fixed") && (curCSSTop == "auto" || curCSSLeft == "auto"),
103 			props = {}, curPosition = {}, curTop, curLeft;
104 
105             if (calculatePosition) {
106                 curPosition = this.position(elem);
107                 curTop = curPosition.top;
108                 curLeft = curPosition.left;
109             } else {
110                 curTop = parseFloat(curCSSTop) || 0;
111                 curLeft = parseFloat(curCSSLeft) || 0;
112             }
113 
114 
115             if (options.top != null) {
116                 props.top = (options.top - curOffset.top) + curTop;
117             }
118             if (options.left != null) {
119                 props.left = (options.left - curOffset.left) + curLeft;
120             }
121             sl.css(elem, props);
122 
123         },
124         /**
125         *获取元素的position属性
126         *@param elem 元素
127         *@return {top:**,left:**}
128         */
129         position: function (elem) {
130 
131             var offsetParent = this.offsetParent(elem), offset = this.getOffset(elem);
132             parentOffset = /^(?:body|html)$/i.test(offsetParent.nodeName) ? { top: 0, left: 0} : this.getOffset(offsetParent);
133 
134             offset.top -= parseFloat(sl.css(elem, "marginTop")) || 0;
135             offset.left -= parseFloat(sl.css(elem, "marginLeft")) || 0;
136 
137             parentOffset.top += parseFloat(sl.css(offsetParent, "borderTopWidth")) || 0;
138             parentOffset.left += parseFloat(sl.css(offsetParent, "borderLeftWidth")) || 0;
139 
140             // (top-parentTop-parentBodrder-maigin)
141             return {
142                 top: offset.top - parentOffset.top,
143                 left: offset.left - parentOffset.left
144             };
145         },
146         /**
147         *@ignore
148         */
149         offsetParent: function (elem) {
150 
151             var offsetParent = elem.offsetParent || document.body;
152             while (offsetParent && (!/^(?:body|html)$/i.test(offsetParent.nodeName) && sl.css(offsetParent, "position") === "static")) {
153                 offsetParent = offsetParent.offsetParent;
154             }
155             return offsetParent;
156         }
157 
158     }
159     var _offset = new offset();
160     /**
161     *设置或者获取元素的位置 相对于页面
162     *@memberOf offset
163     *@param nodes DOM元素
164     *@param value 位置的值{left:,top:} 当为空时候表示获取值
165     *@return {top:**,left:**}
166     */
167     sl.offset = function (nodes, value) {
168         nodes = sl.Convert.convertToArray(nodes, null, sl);
169         if (!nodes.length) return;
170         if (!!value) {
171             for (var i = 0; i < nodes.length; i++) {
172                 _offset.setOffset(nodes[i], value);
173             }
174         }
175         else {
176             return _offset.getOffset(nodes[0]);
177         }
178 
179         return sl.access(nodes, value, null, _offset.getOffset, _offset.setOffset, _offset, null);
180     }
181     sl.position = function (elem) {
182         return _offset.position(elem);
183     };
184     var scrollFun = {};
185     /**
186     *设置或者获取元素的scrollLeft
187     *@memberOf offset
188     @name scrollLeft
189     *@param nodes DOM元素
190     *@param value 位置的值{left:,top:} 当为空时候表示获取值
191     *@return {top:**,left:**}
192     */
193     sl.each(["scrollLeft", "scrollTop"], function (index, name) {
194         scrollFun[name] = function (elem, value) {
195 
196             var win = ("scrollTo" in elem && elem.document) ? elem : (elem.nodeType == 9) ? (elem.defaultView || elem.parentWindow) : false;
197             if (!elem) {
198                 return null;
199             }
200 
201             if (value !== undefined) {
202                 if (win) {
203                     win.scrollTo(
204 						!index ? value : sl.scrollLeft(win),
205 						 index ? value : sl.scrollTop(win)
206 					);
207                 }
208                 else {
209                     elem[name] = value;
210                 }
211             }
212             else {
213                 return win ? ("pageXOffset" in win) ? win[index ? "pageYOffset" : "pageXOffset"] :
214 	(win.document.documentElement[name] ||
215 					win.document.body[name]) :
216 				elem[name];
217 
218 
219             }
220         };
221         sl[name] = function (nodes, value) {
222             nodes = sl.Convert.convertToArray(nodes, null, sl);
223             if (!nodes.length) return;
224             if (!!value) {
225                 for (var i = 0; i < nodes.length; i++) {
226                     scrollFun[name](nodes[i], value);
227                 }
228             }
229             else {
230                 return scrollFun[name](nodes[0]);
231 
232             }
233         };
