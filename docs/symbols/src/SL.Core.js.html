458     *@memberOf sl
459     *@function
460     *@param deep||target,[object1],[objectN]
461     *@example
462     * var object1 = {
463     *        apple: 0,
464     *        banana: { weight: 52, price: 100 },
465     *        cherry: 97
466     *    };
467     *    var object2 = {
468     *        banana: { price: 200 },
469     *        durian: 100
470     *    };
471     *
472     *    console.dir(sl.extend(true, object1, object2));
473     *  {"apple":0,"banana":{"weight":52,"price":200},"cherry":97,"durian":100}
474     *    console.dir(sl.extend(object1, object2));
475     *    {"apple":0,"banana":{"price":200},"cherry":97,"durian":100}
476     */
477     function extend() {
478         var options, name, src, copy, copyIsArray, clone,
479 		target = arguments[0] || {},
480 		i = 1,
481 		length = arguments.length,
482 		deep = false;
483 
484         // Handle a deep copy situation
485         if (typeof target === "boolean") {
486             deep = target;
487             target = arguments[1] || {};
488             i = 2;
489         }
490 
491         // Handle case when target is a string or something (possible in deep copy)
492         if (typeof target !== "object" && (target)) {
493             target = {};
494         }
495         // extend SL itself if only one argument is passed
496         if (length === i) {
497             target = SL;
498             --i;
499         }
500         for (; i < length; i++) {
501             if ((options = arguments[i]) != null) {
502                 for (name in options) {
503                     src = target[name];
504                     copy = options[name];
505                     // Prevent never-ending loop
506                     if (target === copy) {
507                         continue;
508                     }
509                     // Recurse if we're merging plain objects or arrays
510                     if (deep && copy && (SL.InstanceOf.PlainObject(copy) || (copyIsArray = SL.InstanceOf.Array(copy)))) {
511                         if (copyIsArray) {
512                             copyIsArray = false;
513                             clone = src && SL.InstanceOf.Array(src) ? src : [];
514 
515                         } else {
516                             clone = src && SL.InstanceOf.PlainObject(src) ? src : {};
517                         }
518 
519                         // Never move original objects, clone them
520                         target[name] = arguments.callee(deep, clone, copy);
521 
522                         // Don't bring in undefined values
523                     } else if (copy !== undefined) {
524                         target[name] = copy;
525                     }
526                 }
527             }
528         }
529         // Return the modified object
530         return target;

727         *var str = sl.param(params);=>width=100&height=100
728         */
729         this.merge = function (first, second) {
730             var i = first.length, j = 0;
731 
732             if (typeof second.length === "number") {
733                 for (var l = second.length; j < l; j++) {
734                     first[i++] = second[j];
735                 }
736             } else {
737                 while (second[j] !== undefined) {
738                     first[i++] = second[j++];
739                 }
740             }
741 
742             first.length = i;
743 
744             return first;
745         };
746         /**
747         *使用过滤函数过滤数组元素。
748         *@memberOf sl
749         *@function
750         *@name grep
751         *@param array 待过滤数组
752         *@param callback 此函数将处理数组每个元素
753         *@param invert (可选) 如果 "invert" 为 false 或为设置，则函数返回数组中由过滤函数返回 true 的元素，当"invert" 为 true，则返回过滤函数中返回 false 的元素集。
754         *@example
755         *sl.grep( [0,1,2], function(n,i){
756         *  return n > 0;
757         });
758         */
759         this.grep = function (arr, callback, inv) {
760             var ret = [];
761             for (var i = 0, length = arr.length; i < length; i++) {
762                 //若inv为true表示不满足callback条件 inv为false表示满足callback条件
763                 if (!inv !== !callback(arr[i], i)) {
764                     ret.push(arr[i]);
765                 }
766             }
767 
768             return ret;
769         };
770         /**
771         *将一个数组中的元素转换到另一个数组中。
772         *@memberOf sl
773         *@function
774         *@name map
775         *@param array 待转换数组
776         *@param callback 此函数将处理数组每个元素
777         *@example
778         *sl.map( [0,1,2], function(n){
779         *  return n + 4;
780         *   });
781         */
782         this.map = function (arr, callback, arg) {
783             var ret = [], value;
784             for (var i = 0, length = arr.length; i < length; i++) {
785                 value = callback(arr[i], i, arg);
786 
787                 if (value != null) {
788                     ret[ret.length] = value;
789                 }
790             }
791 
792             return ret.concat.apply([], ret);
793         };

838 //判断页面加载
839 SL().create(function (SL) {
840     //事件队列
841     var contentLoadedEventList = [],
842     //是否已经加载完
843     isLoaded = false,
844     //是否已经绑定 防止多次绑定事件
845     isBinded = false,
846     //事件
847     DOMContentLoaded;
848     if (document.addEventListener) {
849         DOMContentLoaded = function () {
850             document.removeEventListener("DOMContentLoaded", DOMContentLoaded, false);
851             fireLoadedEvents();
852         };
853 
854     } else if (document.attachEvent) {
855         DOMContentLoaded = function () {
856             if (document.readyState === "complete") {
857                 document.detachEvent("onreadystatechange", DOMContentLoaded);
858                 fireLoadedEvents();
859             }
860         };
861     }
862 
863     function ready(fn) {
864         BindContentLoadedEvent();
865         if (isLoaded) {
866             fn.call(document);
867         } else if (contentLoadedEventList) {
868             contentLoadedEventList.push(fn);
869         }
870     }
871     function BindContentLoadedEvent() {
872         if (isBinded) {
873             return;
874         }
875         isBinded = true;
876         if (document.readyState === "complete") {
877             return fireLoadedEvents();
878         }
879         if (document.addEventListener) {
880             document.addEventListener("DOMContentLoaded", DOMContentLoaded, false);
881             window.addEventListener("load", fireLoadedEvents, false);
882         } else if (document.attachEvent) {
883             document.attachEvent("onreadystatechange", DOMContentLoaded);
884             window.attachEvent("onload", fireLoadedEvents);
885             var toplevel = false;
886             try {
887                 toplevel = window.frameElement == null;
888             } catch (e) { }
889             if (document.documentElement.doScroll && toplevel) {
890                 doScrollCheck();
891             }
892         }
893     }
894     function fireLoadedEvents() {
895         if (!isLoaded) {
896             if (!document.body) {
897                 return setTimeout(fireLoadedEvents, 13);
898             }
899 
900             isLoaded = true;
901             if (contentLoadedEventList) {
902                 var fn, i = 0;
903                 while ((fn = contentLoadedEventList[i++])) {
904                     fn.call(document);
905                 }
906                 contentLoadedEventList = null;
907             }
908         }
909     }
910     function doScrollCheck() {
911         if (isLoaded) {
912             return;
913         }
914         try {
915             // If IE is used, use the trick by Diego Perini
916             // http://javascript.nwbox.com/IEContentLoaded/
917             document.documentElement.doScroll("left");
918         } catch (error) {
919             setTimeout(doScrollCheck, 1);
920             return;
921         }
922         fireLoadedEvents();
923     }
924     SL.ready = ready;
925 });
