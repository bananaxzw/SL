 63     var oAjaxHelper = {
 64         getData: function (xhr, type) {
 65             /// <summary>
 66             /// 
 67             /// </summary>
 68             /// <param name="xhr"></param>
 69             /// <param name="type">预期options的datatype</param>
 70             /// <returns type=""></returns>
 71             var ct = xhr.getResponseHeader("content-type"),
 72 			xml = type === "xml" || !type && ct && ct.indexOf("xml") >= 0,
 73 			data = xml ? xhr.responseXML : xhr.responseText;
 74 
 75             if (xml && data.documentElement.nodeName === "parsererror") {
 76                 throw new Error("parsererror");
 77             }
 78 
 79             if (typeof data === "string") {
 80                 //用户自定义返回json
 81                 if (type === "json" || !type && ct.indexOf("json") >= 0) {
 82                     if (typeof sl.Josn === "object" && sl.Josn.parse) {
 83                         data = sl.Josn.parse(data);
 84                     } else {
 85                         data = (new Function("return " + data))();
 86                     }
 87                 }
 88                 else if (type === "script" || !type && ct.indexOf("javascript") >= 0) {
 89                     sl.evalSript(data);
 90                 }
 91             }
 92             return data;
 93 
 94         },
 95         isRequestSuccess: function (xhr) {
 96             //IE error sometimes returns 1223 when it should be 204 so treat it as success
 97             try {
 98                 return (!xhr.status && location.protocol == "file:")
 99                     || (xhr.status >= 200 && xhr.status < 300)
100                     || xhr.status === 1223
101                     || (xhr.status == 304)
102                     || (navigator.userAgent.indexOf("Safari") > -1 && typeof xhr.status == "undefined");
103             } catch (e) {
104 
105             }
106             return false;
107 
108         },

110             var _lastModified = xhr.getResponseHeader("Last-Modified"),
111 			_etag = xhr.getResponseHeader("Etag");
112 
113             if (_lastModified) {
114                 lastModified[url] = _lastModified;
115             }
116 
117             if (_etag) {
118                 etag[url] = _etag;
119             }
120 
121             // Opera returns 0 when status is 304
122             return xhr.status === 304 || xhr.status === 0;
123         },
124         setRequestHeaders: function (xhr, option) {

153     /**
154     *ajax请求
155     *@memberOf ajax

157     *@name ajax
158     *@param options  请求参数
159     */
160     this.ajax = function (options) {
161         options = sl.extend(true, {}, sl.ajaxSettting, options);
162         var isComplete = false, status, data, type = options.type.toUpperCase(),
163         xhr = new window.XMLHttpRequest(), jsonp, callbackContext = options.callbackContext || options, noContent = rGETHEAD.test(type);
164         //传进来的data没经过处理
165         if (options.data && options.processData && typeof options.data != "string") {
166             options.data = sl.param(options.data);
167         }
168         // 处理jsonp
169         if (options.dataType === "jsonp") {
170             if (type === "GET") {
171                 if (!jsre.test(options.url)) {
172                     //把url 加上callback=?
173                     options.url += (rquery.test(options.url) ? "&" : "?") + (options.jsonp || "callback") + "=?";
174                 }
175             } else if (!options.data || !jsre.test(options.data)) {
176                 //data后面加callback=?
177                 options.data = (options.data ? options.data + "&" : "") + (options.jsonp || "callback") + "=?";
178             }
179             options.dataType = "json";
180         }
181         //json类型 并且包含callback=?可能要执行跨域操作
182         if (options.dataType === "json" && (options.data && jsre.test(options.data) || jsre.test(options.url))) {
183             jsonp = options.jsonpCallback || ("jsonp" + jsc++);
184             // 把callback=?替换成callback=jsonp
185             if (options.data) {
186                 options.data = (options.data + "").replace(jsre, "=" + jsonp + "$1");
187             }
188             options.url = options.url.replace(jsre, "=" + jsonp + "$1");
189             //当做sript处理
190             options.dataType = "script";
191             // jsonp回调函数 也是成功的处理函数
192             window[jsonp] = window[jsonp] || function (tmp) {
193                 data = tmp;
194                 oHandleEvent.success(options, data, status, xhr);
195                 oHandleEvent.complete(options, xhr, status, data);
196                 window[jsonp] = undefined;
197                 try {
198                     delete window[jsonp];
199                 } catch (e) { }
200 
201                 if (head) {
202                     head.removeChild(script);
203                 }
204             };
205         }
206         //cache (default: true, false for dataType 'script' and 'jsonp')
207         if (options.dataType === "script" && options.cache === null) {
208             options.cache = false;
209         }
210         //是否缓存 GET会有缓存危险 POST不会有 
211         if (options.cache === false && noContent) {
212             var ts = now();
213             //如果已经有时间戳了 替换
214             var ret = options.url.replace(rts, "$1_=" + ts + "$2");
215             //判断是否有时间戳了 没有的话加上时间戳
216             options.url = ret + ((ret === options.url) ? (rquery.test(options.url) ? "&" : "?") + "_=" + ts : "");
217         }
218         //GET 或者 head请求 附加data到url 
219         if (options.data && noContent) {
220             options.url += (rquery.test(options.url) ? "&" : "?") + options.data;
221         }
222         //判断是否跨域
223         var parts = rurl.exec(options.url),
224 			remote = parts && (parts[1] && parts[1] !== location.protocol || parts[2] !== location.host);
225         // jsonp操作 或者从异域加载js
226         if (options.dataType === "script" && type === "GET" && remote) {
227             var head = document.getElementsByTagName("head")[0] || document.documentElement;
228             var script = document.createElement("script");
229             script.src = options.url;
230             if (options.scriptCharset) {
231                 script.charset = options.scriptCharset;
232             }
233             //请求sript并且不是jsonp  因为jsonp里面只有回调函数
234             if (!jsonp) {
235                 var done = false;
236                 // 脚本下载完毕 执行
237                 script.onload = script.onreadystatechange = function () {
238                     if (!done && (!this.readyState ||
239 							this.readyState === "loaded" || this.readyState === "complete")) {
240                         done = true;
241                         oHandleEvent.success(options, data, status, xhr);
242                         oHandleEvent.complete(options, xhr, status, data);
243 
244                         // 防止内存泄露
245                         script.onload = script.onreadystatechange = null;
246                         if (head && script.parentNode) {
247                             head.removeChild(script);
248                         }
249                     }
250                 };
251             }
252             //用insertBefore防止在ie6下head存在base节点时候的bug
253             head.insertBefore(script, head.firstChild);
254             return undefined;
255         }
256         xhr.open(options.type, options.url, options.async);
257 
258         try {
259             if (options.data && options.contentType) {
260                 xhr.setRequestHeader("Content-Type", options.contentType);
261             }
262             xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
263             if (options.ifModified) {
264                 if (lastModified[s.url]) {
265                     xhr.setRequestHeader("If-Modified-Since", lastModified[options.url]);
266                 }
267 
268                 if (etag[options.url]) {
269                     xhr.setRequestHeader("If-None-Match", etag[options.url]);
270                 }
271             }
272 
273         } catch (e) { }
274 
275         var onreadystatechange = xhr.onreadystatechange = function (isTimeout) {
276             //ajax请求被终止
277             if (!xhr || xhr.readyState === 0 || isTimeout === "abort") {
278                 if (!isComplete) {
279                     oHandleEvent.complete(s, xhr, status, data);
280                 }
281 
282                 isComplete = true;
283                 if (xhr) {
284                     xhr.onreadystatechange = function () { };
285                 }
286             }
287             else if (!isComplete && xhr && (xhr.readyState == 4 || isTimeout == "timeout")) {
288                 isComplete = true;
289                 xhr.onreadystatechange = function () { };
290                 status = isTimeout == "timeout" ? "timeout" :
291                 !oAjaxHelper.isRequestSuccess(xhr) ? "error" :
292                 options.ifModified && oAjaxHelper.httpNotModified(xhr, options.url) ? "notmodified" :
293                 "success";
294                 var errorMsg;
295                 if (status == "success" || status === "notmodified") {
296                     try {
297                         data = oAjaxHelper.getData(xhr, options.dataType);
298                     } catch (e) {
299                         status = "parsererror";
300                         errorMsg = e;
301 
302                     }
303                 }
304 
305                 if (status == "success" || status === "notmodified") {
306                     // 因为jsonp里面只有回调函数
307                     if (!jsonp) {
308                         oHandleEvent.success(options, data, status, xhr);
309                     }
310 
311                 } else if (status == "timeout") {
312                     oHandleEvent.timeout(options, xhr, status);
313                 } else {
314                     oHandleEvent.error(options, xhr, status, errorMsg);
315                 }
316                 // Fire the complete handlers
317                 if (!jsonp) {
318                     oHandleEvent.complete(options, xhr, status, data);
319                 }
320                 // 防止内存泄露
321                 if (options.async)
322                     xhr = null;
323             }
324 
325         };
326 
327         // 重写abort ie6支持
328         // Opera不能触发onreadystatechange在abort触发
329         try {
330             var oldAbort = xhr.abort;
331             xhr.abort = function () {
332                 if (xhr) {
333                     //ie7的abort没有call属性
334                     Function.prototype.call.call(oldAbort, xhr);
335                 }
336 
337                 onreadystatechange("abort");
338             };
339         } catch (abortError) { }
340         //异步操作 超时判断
341         if (options.async && options.timeout > 0) {
342             setTimeout(function () {
343                 if (xhr) {
344                     if (!isComplete) {
345                         onreadystatechange("timeout");
346                     }
347                     xhr.abort();
348                 }
349 
350             }, options.timeout);
351 
352         }
353 
354         try {
355             xhr.send(noContent || options.data == null ? null : options.data);
356         } catch (e) {
357 
358         }
359         // firefox 1.5 doesn't fire statechange for sync requests
360         if (!options.async) {
361             onreadystatechange();
362         }
363         return xhr;
364     };

388         sl[_type] = function (url, data, callback, dataType) {
389             // 没有data只有回调函数
390             if (sl.InstanceOf.Function(data)) {
391                 dataType = dataType || callback;
392                 callback = data;
393                 data = null;
394             }
395 
396             return sl.ajax({
397                 type: _type,
398                 url: url,
399                 data: data,
400                 success: callback,
401                 dataType: dataType
402             });
403         };
