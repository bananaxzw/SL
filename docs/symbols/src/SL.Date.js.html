 20     /**
 21     * date扩展
 22     * @class
 23     * @name date
 24     */
 25     var date = function () {
 26         this.LeftPaddingZero = function (source, length) {
 27             var pre = "",
 28         negative = (source < 0),
 29         string = String(Math.abs(source));
 30 
 31             if (string.length < length) {
 32                 pre = (new Array(length - string.length + 1)).join('0');
 33             }
 34 
 35             return (negative ? "-" : "") + pre + string;
 36         }
 37 
 38         this.monthNames = new Array('January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December');
 39         this.monthAbbreviations = new Array('Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec');
 40         this.dayNames = new Array('Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday');
 41         this.dayAbbreviations = new Array('Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat');
 42     };
 43 
 44     date.prototype = {
 45         /**
 46         *格式化时间
 47         *@param {Date} date时间
 48         *@param {String} format格式 日期格式 yyyy：4位数的年 yy:年的后2位  M：正常表示月份 MM：不足10左边补0 MMM:英文表示 NNN英文缩写 d:天数 dd:不足10补充0  EE：英文星期 E英文星期缩写  h:12小时制 H 小时制 m:分钟 mm不足2位补0 s:秒钟 ss不足2位补0
 49         */
 50         format: function (source, pattern) {
 51             if ('string' != typeof pattern) {
 52                 return source.toString();
 53             }
 54 
 55             function replacer(patternPart, result) {
 56                 pattern = pattern.replace(patternPart, result);
 57             }
 58 
 59             var pad = this.LeftPaddingZero;
 60             year = source.getFullYear(),
 61         month = source.getMonth() + 1,
 62         date2 = source.getDate(),
 63         hours = source.getHours(),
 64         minutes = source.getMinutes(),
 65         seconds = source.getSeconds();
 66 
 67             replacer(/yyyy/g, pad(year, 4));
 68             replacer(/yy/g, pad(parseInt(year.toString().slice(2), 10), 2));
 69             replacer(/MM/g, pad(month, 2));
 70             replacer(/M/g, month);
 71             replacer(/dd/g, pad(date2, 2));
 72             replacer(/d/g, date2);
 73 
 74             replacer(/HH/g, pad(hours, 2));
 75             replacer(/H/g, hours);
 76             replacer(/hh/g, pad(hours % 12, 2));
 77             replacer(/h/g, hours % 12);
 78             replacer(/mm/g, pad(minutes, 2));
 79             replacer(/m/g, minutes);
 80             replacer(/ss/g, pad(seconds, 2));
 81             replacer(/s/g, seconds);
 82 
 83             return pattern;
 84         },
 85         parseString: function (source) {
 86             var reg = new RegExp("^\\d+(\\-|\\/)\\d+(\\-|\\/)\\d+\x24");
 87             if ('string' == typeof source) {
 88                 if (reg.test(source) || isNaN(Date.parse(source))) {
 89                     var d = source.split(/ |T/),
 90                 d1 = d.length > 1
 91                         ? d[1].split(/[^\d]/)
 92                         : [0, 0, 0],
 93                 d0 = d[0].split(/[^\d]/);
 94                     return new Date(d0[0] - 0,
 95                             d0[1] - 1,
 96                             d0[2] - 0,
 97                             d1[0] - 0,
 98                             d1[1] - 0,
 99                             d1[2] - 0);
100                 } else {
101                     return new Date(source);
102                 }
103             }
104 
105             return new Date();
106 
107         },
108         isBefore: function (date1, date2) {
109             /// <summary>
110             /// 判断
111             /// </summary>
112             /// <param name="date1"></param>
113             /// <param name="date2"></param>
114             /// <returns type=""></returns>
115             if (date1 === null || date2 === null) return false;
116             return (date1.getTime() < date2.getTime());
117         },
118         /**
119         *将指定的数目（月数，天数，年数等等）加到当前时间上
120         *@param {Date} date时间
121         *@param {String} part 数目的类型（y:年 M月 d天 h小时 m分 s秒）
122         */
123         add: function (date, part, number) {
124 
125             /// <summary>
126             /// 将指定的数目（月数，天数，年数等等）加到当前时间上
127             /// </summary>
128             /// <param name="date">日期</param>
129             /// <param name="part">数目的类型（y:年 M月 d天 h小时 m分 s秒）</param>
130             /// <param name="number"></param>
131             /// <returns type=""></returns>
132             if (typeof (part) == "undefined" || part == null || typeof (number) == "undefined" || number == null) {
133                 return date;
134             }
135             number = +number;
136             if (part == 'y') { // year
137                 date.setFullYear(date.getFullYear() + number);
138             }
139             else if (part == 'M') { // Month
140                 date.setMonth(date.getMonth() + number);
141             }
142             else if (part == 'd') { // Day
143                 date.setDate(date.getDate() + number);
144             }
145             else if (part == 'w') { // Weekday工作日 //没测试
146                 var step = (number > 0) ? 1 : -1;
147                 while (number != 0) {
148                     date.add('d', step);
149                     while (date.getDay() == 0 || date.getDay() == 6) {
150                         arguments.callee('d', step);
151                     }
152                     number -= step;
153                 }
154             }
155             else if (part == 'h') { // Hour
156                 date.setHours(date.getHours() + number);
157             }
158             else if (part == 'm') { // Minute
159                 date.setMinutes(date.getMinutes() + number);
160             }
161             else if (part == 's') { // Second
162                 date.setSeconds(date.getSeconds() + number);
163             }
164             return date;
165 
166         }
167     };
168 
169     SL.Date = SL.Date || {};
170     SL.Date = new date();
