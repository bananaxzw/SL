 16 sl.create(function () {
 17 
 18     /**
 19     *查询字符串帮助
 20     *@class
 21     *@name QueryString
 22     *@param {String} str 查询字符串的值
 23     *@example
 24     *window.onload = function () {
 25     *     var quering = new sl.QueryString("jj.aspx?name=xuzhiwei&age=11");
 26     *     console.log(quering.name);
 27     *     console.log(quering.age);
 28     *     console.log(quering.toString());
 29     * }
 30     */
 31     var QueryString = function (str) {
 32         this.string = str;
 33         this.length = 0;
 34         var re = /[?&]([^=]+)=([^&]*)/g;
 35         var match = re.exec(str);
 36         while (match) {
 37             var name = match[1];
 38             var value = decodeURI(match[2]);
 39             this[name] = value;
 40             this[this.length] = new QueryString.Pair(name, value);
 41             this.length++;
 42             match = re.exec(str);
 43         }
 44     };
 45     /**
 46     *@ignore
 47     */
 48     QueryString.prototype = {
 49         /**
 50         *@ignore
 51         */
 52         toString: function () {
 53             return this.string;
 54         }
 55     };
 56     /**
 57     *@ignore
 58     */
 59     QueryString.Pair = function (name, value) {
 60         this.name = name;
 61         this.value = value;
 62     };
 63     /**
 64     *@ignore
 65     */
 66     QueryString.Pair.prototype.toString = function () {
 67         return this.name + '=' + this.value;
 68     };
 69     sl.QueryString = function (str) {
 70         return new QueryString(str);
 71     }
 72 });
