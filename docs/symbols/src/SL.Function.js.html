 25     SLFunction.prototype = {
 26         /**
 27         *方法委托 转移方法的所有对象
 28         *@Param {Object} oThis 对象
 29         *@Param {Function} fun 方法
 30         *@returns {Function} 新的代理方法
 31         *@example
 32         * function p() {
 33         *    alert(this.name);
 34         *}
 35         *var person =new  function () {
 36         *    this.name = "xuzhiwei";
 37         *}
 38         * var mm=  SL().Function.createDelegate(person, p);
 39         * mm(); //xuzhiwei
 40         */
 41         createDelegate: function (oThis, fun) {
 42             return function () {
 43                 return fun.apply(oThis, arguments)
 44             }
 45         }
 46     };
 47     SL.Function = new SLFunction();
