 23 sl.create(function () {
 24     var valHooks = {
 25         option: {
 26             get: function (elem) {
 27                 //看看有没有设置value ie没设置value默认会""
 28                 var val = elem.attributes.value;
 29                 return !val || val.specified ? elem.value : elem.text;
 30             }
 31         },
 32         select: {
 33             get: function (elem) {
 34                 var value, i, max, option,
 35 					index = elem.selectedIndex,
 36 					values = [],
 37 					options = elem.options,
 38 					one = elem.type === "select-one";
 39 
 40                 if (index < 0) {
 41                     return null;
 42                 }
 43                 i = one ? index : 0;
 44                 max = one ? index + 1 : options.length;
 45                 for (; i < max; i++) {
 46                     option = options[i];
 47 
 48                     // 某些浏览器在select设置disabled时候选项也disbled 要让他们不disabled
 49                     if (option.selected && (sl.Support.optDisabled ? !option.disabled : option.getAttribute("disabled") === null) &&
 50 							(!option.parentNode.disabled || !/optgroup/i.test(option.parentNode.nodeName))) {
 51                         value = sl.attr.getValue(option);
 52                         if (one) {
 53                             return value;
 54                         }
 55                         values.push(value);
 56                     }
 57                 }
 58 
 59                 //  select.val() broken in IE after form.reset()
 60                 if (one && !values.length && options.length) {
 61                     return sl.attr.getValue(options[index]);
 62                 }
 63 
 64                 return values;
 65             },
 66 
 67             set: function (elem, value) {
 68                 var values = sl.Convert.convertToArray(value);
 69 
 70                 for (var i = 0, length = elem.options.length; i < length; i++) {
 71                     var opt = elem.options[i];
 72                     if (sl.Array.indexOf(values, sl.attr.getValue(opt)) > -1) {
 73                         opt.selected = true;
 74                     }
 75                 }
 76 
 77                 if (!values.length) {
 78                     elem.selectedIndex = -1;
 79                 }
 80                 return values;
 81             }
 82         }
 83     };
 84     // Radios and checkboxes getter/setter
 85     if (!sl.Support.checkOn) {
 86         sl.each(["radio", "checkbox"], function (i, d) {
 87             valHooks[d] = {
 88                 get: function (elem) {
 89                     //某些浏览器 webkit没设置radio和checkbox的值时候value为空 而ie和FF为on 统一为on
 90                     return elem.getAttribute("value") === null ? "on" : elem.value;
 91                 },
 92                 set: function (elem, value) {
 93                     var values = sl.Convert.convertToArray(value);
 94 
 95                     return (elem.checked = sl.Array.indexOf(values, sl.attr.getValue(elem)) >= 0);
 96 
 97                 }
 98 
 99             };
100         });
101     }

103     /**
104     * @description DOM属性操作
105     * @class DOM属性操作
106     * @name attribute
107     */
108     function attribute() { }
109     attribute.prototype = {
110         /**
111         *获取元素属性
112         *@param  ele DOM元素
113         *@param  name 属性名
114         *@return 属性值
115         */
116         getAttr: function (ele, name) {
117             if (/href|src|width|height|colSpan|rowSpan/.test(name)) {
118                 /**IE的getAttribute支持第二个参数，可以为 0,1,2,4
119                 0 是默认；1 区分属性的大小写；2取出源代码中的原字符串值(注，IE67对动态创建的节点没效)。
120                 IE 在取 href 的时候默认拿出来的是绝对路径，加参数2得到我们所需要的相对路径。*/
121                 return ele.getAttribute(name, 2);
122             } else if ("style" === name) {
123                 return ele.style.cssText;
124             } else if (name == "tabIndex") {
125                 var attributeNode = ele.getAttributeNode("tabIndex");
126                 return attributeNode && attributeNode.specified
127 						? attributeNode.value
128 						: ele.nodeName.match(/^(a|area|button|input|object|select|textarea)$/i)
129 							? 0
130 							: undefined;
131             } else {
132                 return ele.getAttribute(name);
133             }
134         },
135         /**
136         *设置元素属性
137         *@param  ele DOM元素
138         *@param  name 属性名
139         *@param  value 属性值
140         */
141         setAttr: function (ele, name, value) {
142             //设置属性
143             if (value == null) {
144                 ele.removeAttribute(name);
145             } else {
146                 if ("style" === name) {
147                     ele.style.cssText ? ele.style.cssText = value : ele.setAttribute("style", value);
148                 } else {
149                     ele.setAttribute(name, value);
150                 }
151             }
152 
153         },
154         /**
155         *移除元素属性
156         *@param  ele DOM元素
157         *@param  name 属性名
158         */
159         removeAttr: function (ele, name) {
160             this.setAttr(ele, name, "");
161             if (ele.nodeType === 1) {
162                 if (ele.removeAttribute) {
163                     ele.removeAttribute(name);
164                 }
165                 else if (ele.attributes && ele.attributes.removeNamedItem) {
166                     ele.attributes.removeNamedItem(name);
167                 }
168             }
169 
170         },
171         /**
172         *添加元素CLASS属性
173         *@param  ele DOM元素
174         *@param  value CLASS名字
175         */
176         addClass: function (ele, value) {
177             if (value && typeof value === "string") {
178                 //分割
179                 var classNames = (value || "").split(/\s+/);
180                 if (ele.nodeType === 1) {
181                     if (!ele.className && classNames.length === 1) {
182                         ele.className = value;
183 
184                     } else {
185                         var className = " " + ele.className + " ";
186                         for (var c = 0, cl = classNames.length; c < cl; c++) {
187                             if (className.indexOf(" " + classNames[c] + " ") < 0) {
188                                 ele.className += " " + classNames[c];
189                             }
190                         }
191                     }
192                 }
193             }
194 
195         },
196         /**
197         *是否具有某CLASS
198         *@param  ele DOM元素
199         *@param  value CLASS名字
200         *@return boolean
201         */
202         hasClass: function (ele, value) {
203             var re = new RegExp('(\\s|^)' + value + '(\\s|$)');
204             return re.test(ele.className.replace(/[\n\t]/, " "));
205 
206         },
207         /**
208         *移除某个CLASS
209         *@param  ele DOM元素
210         *@param  value CLASS名字
211         */
212         removeClass: function (ele, value) {
213             if ((value && typeof value === "string") || value === undefined) {
214                 var classNames = (value || "").split(/\s+/);
215 
216                 if (ele.nodeType === 1 && ele.className) {
217                     if (value) {
218                         var className = (" " + ele.className + " ").replace(/[\n\t]/g, " ");
219                         for (var c = 0, cl = classNames.length; c < cl; c++) {
220                             className = className.replace(" " + classNames[c] + " ", " ");
221                         }
222                         ele.className = className.substring(1, className.length - 1);
223 
224                     } else {
225                         ele.className = "";
226                     }
227                 }
228             }
229         },
230         /**
231         *触发和移除某个CLASS
232         *@param  ele DOM元素
233         *@param  value CLASS名字
234         */
235         toggleClass: function (ele, value) {
236             if (this.hasClass(ele, value)) {
237                 this.removeClass(ele, value);
238             } else {
239                 this.addClass(ele, value);
240             }
241         },
242         /**
243         *获取元素ele的Value值
244         *@param  ele DOM元素
245         *@return value值
246         */
247         getValue: function (ele) {
248             var hooks = valHooks[ele.type] || valHooks[ele.nodeName.toLowerCase()];
249             if (hooks && "get" in hooks && (ret = hooks.get(ele)) !== undefined) {
250                 return ret;
251             }
252             if (sl.InstanceOf.DOMElement(ele) && "value" in ele) {
253                 return ele.value;
254             }
255             return "";
256         },
257         /**
258         *设置元素value值
259         *@param  ele DOM元素
260         *@param  value value值
261         */
262         setValue: function (ele, value) {
263             if (ele.nodeType != 1) {
264                 return;
265             }
266             var hooks = valHooks[ele.type] || valHooks[ele.nodeName.toLowerCase()];
267             if (!hooks || !("set" in hooks) || hooks.set(ele, value) === undefined) {
268                 if ("value" in ele) {
269                     ele.value = value;
270                 }
271             }
272 
273         }
274     }
275     sl.attr = new attribute();
276 });
