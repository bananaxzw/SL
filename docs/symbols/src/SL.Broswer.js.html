 33         function fixedVersion(ver, floatLength) {
 34             ver = ("" + ver).replace(/_/g, ".");
 35             floatLength = floatLength || 1;
 36             ver = String(ver).split(".");
 37             ver = ver[0] + "." + (ver[1] || "0");
 38             ver = Number(ver).toFixed(floatLength);
 39             return ver;
 40 
 41         }
