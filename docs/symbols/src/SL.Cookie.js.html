 16 sl.create(function () {
 17     /**
 18     * @class
 19     * @name Cookie
 20     * @example
 21     * sl.Cookie.set("name1", "xuzhiwei", ".testxuzhiwei11.com", "/", 1);
 22     * console.log(sl.Cookie.get("name1"));
 23     * sl.Cookie.remove("name1");
 24     * console.log(sl.Cookie.get("name1"));
 25     */
 26     var Cookie = function () { };
 27     Cookie.prototype = {
 28         /**
 29          * 获取指定名称的cookie值
 30          * 
 31          * @param {String} name cookie名称
 32          * @return {String} 获取到的cookie值
 33          */
 34         get: function (name) {
 35             /// <summary>
 36             /// 获取cookie
 37             /// </summary>
 38             /// <param name="name"></param>
 39             /// <returns type=""></returns>
 40             var r = new RegExp("(?:^|;+|\\s+)" + name + "=([^;]*)");
 41             var m = window.document.cookie.match(r);
 42             return (!m ? "" : m[1]);
 43         },
 44         /**
 45         * 设置一个cookie
 46         * @param {String} name cookie名称
 47         * @param {String} value cookie值
 48         * @param {String} domain 所在域名
 49         * @param {String} path 所在路径
 50         * @param {Number} hour 存活时间，单位:小时
 51         * @param {Boolean} secure 是否采用安全传输
 52         * @return {Boolean} 是否成功
 53         */
 54         set: function (name, value, domain, path, hour,secure) {
 55             /// <summary>
 56             /// 设置cookie
 57             /// </summary>
 58             /// <param name="name"></param>
 59             /// <param name="value"></param>
 60             /// <param name="domain"></param>
 61             /// <param name="path"></param>
 62             /// <param name="hour"></param>
 63             /// <returns type=""></returns>
 64             if (hour) {
 65                 var today = new Date();
 66                 var expire = new Date();
 67                 expire.setTime(today.getTime() + 3600000 * hour);
 68             }
 69 
 70             var string = name + "=" + value + "; " + (hour ? ("expires=" + expire.toGMTString() + "; ") : "") + (path ? ("path=" + path + "; ") : "path=/; ") + (domain ? ("domain=" + domain + ";") : ("domain=" + window.location.host + ";")) + (secure ? " secure" : "");
 71             window.document.cookie =string;
 72             return true;
 73         },
 74         /**
 75         * 删除指定cookie,复写为过期
 76         * 
 77         * @param {String} name cookie名称
 78         * @param {String} domain 所在域
 79         * @param {String} path 所在路径
 80         */
 81         remove: function (name, domain, path) {
 82             /// <summary>
 83             /// 移除cookie
 84             /// </summary>
 85             /// <param name="name"></param>
 86             /// <param name="domain"></param>
 87             /// <param name="path"></param>
 88             window.document.cookie = name + "=; expires=Mon, 26 Jul 1997 05:00:00 GMT; " + (path ? ("path=" + path + "; ") : "path=/; ") + (domain ? ("domain=" + domain + ";") : ("domain=" + window.location.host + ";"));
 89         }
 90     };
 91 
 92     sl.Cookie = SL.Cookie || {};
 93     sl.Cookie = new Cookie();
 94 });
