 25     function chain(selector, context) {
 26         // (""), (null), or (undefined)
 27         if (!selector) {
 28             return this;
 29         }
 30         // (DOMElement)
 31         if (selector.nodeType || sl.InstanceOf.Window(selector)) {
 32             //            this.context = selector;
 33             this.elements = [selector];
 34             return this;
 35         }
 36         if (context) {
 37             if (this.isChain(context)) {
 38                 this.elements = innerFind(selector, context.elements);
 39             } else {
 40                 if (sl.InstanceOf.DOMElement(context)) {
 41                     this.elements = sl.select(selector, context);
 42                 } else if (sl.InstanceOf.DOMNodeList(context)) {
 43                     this.elements = innerFind(selector, context);
 44                 } else {
 45                     this.elements = [];
 46                 }
 47             }
 48         }
 49         else {
 50             if (typeof selector === "string") {
 51                 //处理html
 52                 if (selector.charAt(0) === "<" && selector.charAt(selector.length - 1) === ">" && selector.length >= 3) {
 53                     this.elements = sl.Convert.convertToArray(sl.Dom.parseHtml(selector).childNodes, "", this);
 54 
 55                 } else {
 56                     //正常的选择器
 57                     this.elements = sl.select(selector);
 58                 }
 59             } else if (sl.InstanceOf.DOMElement(selector)) {
 60                 this.elements = [selector];
 61             } else if (sl.InstanceOf.DOMNodeList(selector)) {
 62                 this.elements = sl.Convert.convertToArray(selector);
 63             } else if (this.isChain(selector)) {
 64                 this.elements = [];
 65                 for (var i = 0; i < selector.elements.length; i++) {
 66                     this.elements.push(selector.elements[i]);
 67                 }
 68             }
 69         }
 70         this.length = this.elements.length;
 71     }
