 16 SL().create(function (SL) {

 23     var number = function () { };
 24     number.prototype = {
 25         /**
 26         *去除非数字字符   比如3df12335.52saa return 312335.52  保留-  .
 27         */
 28         stripNonNumeric: function (str) {
 29             /// <summary>
 30             /// 去除非数字字符   比如3df12335.52saa return 312335.52  保留-  .
 31             /// </summary>
 32             /// <param name="str"></param>
 33             /// <returns type=""></returns>
 34             str += '';
 35             var rgx = /^\d|\.|-$/;
 36             var out = '';
 37             for (var i = 0; i < str.length; i++) {
 38                 if (rgx.test(str.charAt(i))) {
 39                     if (!((str.charAt(i) == '.' && out.indexOf('.') != -1) ||
 40              (str.charAt(i) == '-' && out.length != 0))) {
 41                         out += str.charAt(i);
 42                     }
 43                 }
 44             }
 45             return out;
 46 
 47         },
 48         /** 
 49         *格式化数字
 50         *@param {number} num
 51         *@param {string} format 格式       
 52         *@example 
 53         *‘0' - (123456) 表示没有不保留小数位
 54         * ‘0.00′ - (123456.78) 保留2位小数位
 55         *‘0.0000′ - (123456.7890) 保留4位小数位 不够的右端补充0
 56         *‘0,000.00′ - (123,456.78) 保留2位小数位 并且有冒号隔开
 57         */
 58         format: function (num, format) {
 59             /// <summary>
 60             /// 格式化数字
 61             /// ‘0' - (123456) 表示没有不保留小数位
 62             ///‘0.00′ - (123456.78) 保留2位小数位
 63             ///‘0.0000′ - (123456.7890) 保留4位小数位 不够的右端补充0
 64             ///‘0,000.00′ - (123,456.78) 保留2位小数位 并且有冒号隔开
 65             /// </summary>
 66             /// <param name="num"></param>
 67             /// <param name="format"></param>
 68             /// <returns type=""></returns>
 69 
 70            // .replace(/(\d{1,2})(?=(\d{3})+\b)/g,"$1,")
 71 
 72             var hasComma = -1 < format.indexOf(','),
 73     psplit = this.stripNonNumeric(format).split('.'),
 74     that = Number(this.stripNonNumeric(num));
 75 
 76             if (1 < psplit.length) {
 77                 that = that.toFixed(psplit[1].length);
 78             }
 79             else if (2 < psplit.length) {
 80                 throw Error('无效的格式符:' + format);
 81             }
 82             else {
 83                 that = that.toFixed(0);
 84             }
 85 
 86             var fnum = that.toString();
 87 
 88             if (hasComma) {
 89                 psplit = fnum.split('.');
 90 
 91                 var cnum = psplit[0],
 92       parr = [],
 93       j = cnum.length,
 94       m = Math.floor(j / 3),
 95       n = cnum.length % 3 || 3;
 96                 for (var i = 0; i < j; i += n) {
 97                     if (i != 0) { n = 3; }
 98                     parr[parr.length] = cnum.substr(i, n);
 99                     m -= 1;
100                 }
101                 fnum = parr.join(',');
102                 if (psplit[1]) { fnum += "." + psplit[1]; }
103             }
104 
105             return format.replace(/[\d,?\.?]+/, fnum);
106 
107 
108         }
109     }

111     SL.Number = new number();
