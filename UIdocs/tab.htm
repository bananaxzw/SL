    <script type="text/javascript">
        var tt;
        sl.ready(function () {
            tt = new sl.ui.tab(document.getElementById("tt"));
        })
        var index = 0;
        function addTab() {
            index++;
            tt.add({
                title: 'New Tab ' + index,
                content: 'Tab Body ' + index,
                iconCls: 'icon-save',
                closable: true
            });
        }
        function getSelected() {
            var tab = tt.getSelected();
            alert('Selected: ' + tab.title);
            alert('Selected: ' + tab.index);
        }
        function SelectedByIndex3() {
            tt.selectAt(3);

        }
        function SelectedByIndex4() {
            tt.selectAt(4);
        }
    
    </script>

名称 说明
title 表示选项卡的名称
closable 是否显示关闭按钮
iconcls tab标签的图标
selected 默认选中

名称 说明
add({ title: '', content: '', icon: null,closable: false, selected: true}) 添加一个tab
getSelected 获取选中的标签信息 返回格式 {id: id,title: title,isSelect: boolean,index: index}
selectAt(num) 选中指定的索引标签 从0开始索引
